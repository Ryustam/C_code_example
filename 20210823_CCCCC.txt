#pragma warning(disable:4996)
#include <stdio.h>

int main() {

	char str[50];
	int i = 0;
	int code;
	char temp;


	scanf("%s", str);

	while (str[i] != '\0') {
		i++;
	}

	code = i-1;

	for (i = 0; i <= code / 2; i++) {
		temp = str[i];
		str[i] = str[code - i];
		str[code - i] = temp;
	}

	for (i = 0; i<code+1; i++)
		printf("%c", str[i]);

	return 0;
}
